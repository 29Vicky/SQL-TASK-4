CREATE DATABASE JOBS;
USE JOBS;
CREATE TABLE LIST(
JOB_ID INT PRIMARY KEY,
Name VARCHAR(50),
SALARY INT(45)
);
INSERT INTO LIST
( JOB_ID,Name,SALARY)

;


SELECT  SUM(SALARY) AS total_salary
FROM LIST; 
 
SELECT SALARY,COUNT(*) AS SALARY_COUNT
FROM LIST
GROUP BY SALARY;


SELECT AVG(SALARY) AS avg_SALARY
FROM LIST;


SELECT SALARY,
       COUNT(*) AS num_SALARY,
       AVG(SALARY) AS avg_SALARY
FROM LIST
GROUP BY SALARY;
